## 题目描述

[977. 有序数组的平方](https://leetcode-cn.com/problems/squares-of-a-sorted-array/)

难度简单

给你一个按 **非递减顺序** 排序的整数数组 `nums`，返回 **每个数字的平方** 组成的新数组，要求也按 **非递减顺序** 排序。

 <!-- more -->

**示例 1：**

```
输入：nums = [-4,-1,0,3,10]
输出：[0,1,9,16,100]
解释：平方后，数组变为 [16,1,0,9,100]
排序后，数组变为 [0,1,9,16,100]
```

**示例 2：**

```
输入：nums = [-7,-3,2,3,11]
输出：[4,9,9,49,121]
```

**提示：**

- `1 <= nums.length <= 104`
- `-104 <= nums[i] <= 104`
- `nums` 已按 **非递减顺序** 排序

**进阶：**

- 请你设计时间复杂度为 `O(n)` 的算法解决本问题

## 解题思路及代码

```go
// 暴力做法
func sortedSquares(nums []int) []int {
    // 取绝对值，然后使用带flag的冒泡排序
    for i:=0; i<len(nums)-1; i++ {
        flag := false
        for j:=0; j<len(nums)-1-i; j++ {
            if Abs(nums[j]) > Abs(nums[j+1]) {
                nums[j], nums[j+1] = nums[j+1], nums[j]
                flag = true
            }
        }
        if !flag { break }
    }
    // 取平方
    ret := make([]int, len(nums))
    for i:=0; i<len(nums); i++ {
        ret[i] = nums[i] * nums[i]
    }
    return ret
}

func Abs(i int) int {
    if i < 0 {
        return -1*i
    }
    return i
}

// 利用原本有序规律（负数段平方后逆序，正数段平方后正序），找出临界点，然后归并排序
func sortedSquares(nums []int) []int {
    // 找出界限
    bound := -1                 // 这里 如果全为正则为-1，全为负责为n
    for i:=0; i<len(nums)-1 && nums[i] < 0; i++ {
        bound = i
    }
    pow(nums)
    // 归并, 注意负数段的有序是反的
    temp := make([]int, 0)
    merge := func (l, mid, r int) {
        p1, p2 := mid, mid+1
        for p1>=l && p2<=r {
            if nums[p1] < nums[p2] {
                temp = append(temp, nums[p1])
                p1--
            }else {
                temp = append(temp, nums[p2])
                p2++
            }
        }
        for p1 >= l{
            temp = append(temp, nums[p1])
            p1--
        }
        for p2 <= r {
            temp = append(temp, nums[p2])
            p2++
        }
        for i:=0; i<len(temp); i++ {
            nums[i+l] = temp[i]
        }
    }
    
    merge(0, bound, len(nums)-1)
    return nums
}

func pow(nums []int) {
    for i, v := range nums {
        nums[i] = v*v
    }
}

// 直接双指针比较平方，大的放后面
func sortedSquares(nums []int) []int {
    ans := make([]int, len(nums))
    p1, p2, t := 0, len(nums)-1, len(nums)-1
    for p1 <= p2 {
        if nums[p1] * nums[p1] > nums[p2] * nums[p2] {
            ans[t] = nums[p1] * nums[p1]
            p1++
        }else {
            ans[t] = nums[p2] * nums[p2]
            p2--
        }
        t--
    }
    return ans
}
```
