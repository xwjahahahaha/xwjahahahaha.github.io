## 题目描述

[76. 最小覆盖子串](https://leetcode-cn.com/problems/minimum-window-substring/)

难度困难

给你一个字符串 `s` 、一个字符串 `t` 。返回 `s` 中涵盖 `t` 所有字符的最小子串。如果 `s` 中不存在涵盖 `t` 所有字符的子串，则返回空字符串 `""` 。

 <!-- more -->

**注意：**

- 对于 `t` 中重复字符，我们寻找的子字符串中该字符数量必须不少于 `t` 中该字符数量。
- 如果 `s` 中存在这样的子串，我们保证它是唯一的答案。

 

**示例 1：**

```
输入：s = "ADOBECODEBANC", t = "ABC"
输出："BANC"
```

**示例 2：**

```
输入：s = "a", t = "a"
输出："a"
```

**示例 3:**

```
输入: s = "a", t = "aa"
输出: ""
解释: t 中两个字符 'a' 均应包含在 s 的子串中，
因此没有符合条件的子字符串，返回空字符串。
```

 

**提示：**

- `1 <= s.length, t.length <= 105`
- `s` 和 `t` 由英文字母组成

 

**进阶：**你能设计一个在 `o(n)` 时间内解决此问题的算法吗？



## 解题思路及代码

```go
// 滑动窗口
// 注意：这一题是当窗口中间状态一旦不满足（可能是缩小窗口导致）就立即滑动右边直到让其满足
func minWindow(s string, t string) string {
    ansL, minLen := -1, math.MaxInt32
    // 1. 将t转换为一个tMap, sMap记录当前覆盖程度
    sMap, tMap := make(map[byte]int, 0), make(map[byte]int, 0)
    for _, c := range t {
        tMap[byte(c)]+=1
    }
    // 2. 滑动窗口[l, r]，一旦满足就
    for l, r := 0, 0; r < len(s); r++ { // 创建右端不断右移
        // 判断当前字符是否覆盖
        if _, has := tMap[s[r]]; has {
            sMap[s[r]]+=1       // 存在则记录
        } 
        // 判断是否要缩减窗口
        for check(sMap, tMap) && l <= r {
            // 比较最小覆盖长度
            if (r-l+1) < minLen {
                minLen = r-l+1
                ansL = l
            }
            // 开始缩减
            if _, has := tMap[s[l]]; has {
                // 如果有就删除记录
                sMap[s[l]]-=1
            }
            l++
        }
    }
    if ansL == -1 {
        return ""
    }
    return s[ansL:ansL+minLen]
}

// 检查当前s子串是否覆盖t
func check(sMap, tMap map[byte]int) bool {
    for k, v := range tMap {
        if sMap[k] < v {
            return false
        }
    }
    return true
}
```

```go
// 滑动窗口
func minWindow(s string, t string) string {
    //1. 转换t为map
    tRecord, sRecord := make(map[byte]int), make(map[byte]int)
    for i:=0; i<len(t); i++ {
        tRecord[t[i]]++
    }

    l, r, n, minLen, start := 0, 0, len(s), math.MaxInt, -1
    for r < n {
        // 移动右边界
        // 判断是否是t中出现的字符
        if tRecord[s[r]] > 0 {
            sRecord[s[r]]++
        }
        r++
    
        // 判断是否需要缩小左边界
        for l <= r && compareMap(sRecord, tRecord) {    // 只有覆盖才会缩小左边界
            // 判断结果
            if size := r-l; size < minLen {
                minLen = size
                start = l
            }
            // 缩小到排除掉第一个覆盖字符
            sRecord[s[l]]--
            l++
        }
    }

    // 需要判断一下是否经过了判断比较最小
    if start == -1 {
        // 没经过比较就说明未开始收缩左边就已经结束, s没完全覆盖t
        return ""
    }

    return s[start:start+minLen]
}

func compareMap(a, b map[byte]int) bool {
    for k := range b {
        if a[k] < b[k] {
            return false
        }
    }
    
   return true
}
```

三刷

```go
func minWindow(s string, t string) string {
    countTMap, countSMap := make(map[byte]int), make(map[byte]int)
    // 统计t
    for i:=0; i<len(t); i++ {
        countTMap[t[i]] ++
    }
    // 遍历s
    l, r := 0, 0
    // 注意：minLen初始化设置最大值而不是len(s), 因为当s长度为1比较最小值时写的<那么就不会赋值给minLen和start（所以也可以使用len(s)但是比较修改为<=）
    start, minLen := -1, math.MaxInt        
    for r < len(s) {
        // 扫描右边界当前字符
        // 判断是否是t中存在的字符，如果是则记录
        if countTMap[s[r]] > 0 {
            countSMap[s[r]]++
        }
        // 滑动窗口
        // 注意此处l可以等于r，当一个字符的时候也要考虑
        for l <= r && isCover(countSMap, countTMap) {
            // 比较最小值
            if r-l+1 < minLen {
                minLen = r-l+1
                start = l
            }
            // 移动左边界
            countSMap[s[l]]--
            l++     // 注意后++
        }
        r++
    }
    if start == -1 {
        // 说明s无法覆盖t，所以直接返回空串
        return ""
    }
    return s[start:start+minLen]
}

// 判断c1是否覆盖c2
func isCover(c1, c2 map[byte]int) bool {
    if len(c1) < len(c2) {
        return false
    }
    for k, v := range c2 {
        if t, has := c1[k]; !has {
            return false
        }else if t < v {
            return false
        } 
    }
    return true
}
```
