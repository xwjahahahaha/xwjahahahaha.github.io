## 题目描述

[143. 重排链表](https://leetcode-cn.com/problems/reorder-list/)

难度中等

给定一个单链表 `L` 的头节点 `head` ，单链表 `L` 表示为：

```
L0 → L1 → … → Ln - 1 → Ln
```

请将其重新排列后变为：

```
L0 → Ln → L1 → Ln - 1 → L2 → Ln - 2 → …
```

不能只是单纯的改变节点内部的值，而是需要实际的进行节点交换。

## 解题思路及代码

```go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */

// 整体思路，三步：
// 1. 找到链表中点
// 2. 翻转右半部分
// 3. 左右两边合并（不是按顺序，而是节点依次合并）
func reorderList(head *ListNode)  {
    midNode := findMidNode(head)
    rightNode := filpList(midNode.Next)
    midNode.Next = nil      // 切断两边
    mergeListNode(head, rightNode)
}

// 两次遍历，第一次记录长度，第二次根据长度遍历到中点
func findMidNode(head *ListNode) *ListNode {
    p, n := head, 0
    for p != nil {
        n++
        p = p.Next
    }
    p = head
    for i:=0; i<n/2; i++ {
        p = p.Next
    }
    return p
}

// 翻转链表
func filpList(head *ListNode) *ListNode {
    if head == nil || head.Next == nil {
        return head
    }
    l, f := head, head.Next
    for f != nil {
        t := f.Next
        f.Next = l
        l = f
        f = t
    }
    head.Next = nil
    return l
}

// 合并链表(不需要按序)
func mergeListNode(head1, head2 *ListNode) {
    p1, p2, p, tag := head1, head2, &ListNode{}, 1
    for p1 != nil && p2 != nil {
        if tag == 1 {
            p.Next = p1
            p1 = p1.Next
        }else {
            p.Next = p2
            p2 = p2.Next
        }
        tag *= -1
        p = p.Next
    }
    // 处理剩余
    if p1 != nil {
        p.Next = p1
    }
    if p2 != nil {
        p.Next = p2
    }
}
```

/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
// 找到链表中点
// 翻转右侧节点，然后合并
func reorderList(head *ListNode)  {
    if head == nil || head.Next == nil {
        return
    }

    // 找到中点
    mid := getMid(head)

    p1, p2 := head, mid.Next

    // 断开
    mid.Next = nil

    // 翻转后半部分
    p2 = filp(p2)
    
    // 合并两个链表(不用按大小)
    tmpHead, flag := &ListNode{}, 1
    p := tmpHead
    for p1 != nil &&  p2 != nil {
        if flag == 1 {
            p.Next = p1
            p1 = p1.Next
        }else {
            p.Next = p2
            p2 = p2.Next
        }

        flag *= -1
        p = p.Next
    } 

    if p1 != nil {
        p.Next = p1
    }
    if p2 != nil {
        p.Next = p2
    }
}

func getMid(head *ListNode) *ListNode {
    sum := 0 
    for p:=head; p!=nil; p=p.Next { sum++ }
    half := sum/2
    if sum % 2 == 0 {
        half--
    }
    p := head
    for i:=0; i<half; i++ {
        p = p.Next
    }

    return p
}

func filp(head *ListNode) *ListNode {
    if head == nil || head.Next == nil {
        return head
    }

    s, f := head, head.Next
    for f != nil {
        t := f.Next
        f.Next = s
        s = f
        f = t
    }

    head.Next = nil 
    return s
}
